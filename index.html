    <script>

        // Load data on start

        window.onload = () => {

            const savedData = JSON.parse(localStorage.getItem('exploreData')) || { views: 0, reach: 0, followers: 0 };

            document.getElementById('input-views').value = savedData.views;

            document.getElementById('input-reach').value = savedData.reach;

            document.getElementById('input-followers').value = savedData.followers;

            updateUI(savedData);

        };


        function toggleAdmin() {

            const panel = document.getElementById('admin-panel');

            panel.style.display = panel.style.display === 'block' ? 'none' : 'block';

        }


        function saveStats() {

            const data = {

                views: parseInt(document.getElementById('input-views').value) || 0,

                reach: parseInt(document.getElementById('input-reach').value) || 0,

                followers: parseInt(document.getElementById('input-followers').value) || 0

            };

            localStorage.setItem('exploreData', JSON.stringify(data));

            updateUI(data);

            toggleAdmin();


            if (data.followers >= 10000) {

                confetti({ particleCount: 150, spread: 70, origin: { y: 0.6 }, colors: ['#39FF14', '#ffffff'] });

            }

        }


        function updateUI(data) {

            animateNumber('display-views', data.views);

            animateNumber('display-reach', data.reach);

            animateNumber('display-followers', data.followers);


            // Analogy Logic (20,000 per court)

            const viewMult = Math.floor(data.views / 20000);

            const reachMult = Math.floor(data.reach / 20000);


            document.getElementById('mult-views').innerText = 'x' + viewMult.toLocaleString();

            document.getElementById('text-views').innerText = viewMult.toLocaleString();

            document.getElementById('mult-reach').innerText = 'x' + reachMult.toLocaleString();

            document.getElementById('text-reach').innerText = reachMult.toLocaleString();


            // Progress Bar Logic

            const progress = Math.min((data.followers / 10000) * 100, 100);

            document.getElementById('goal-fill').style.width = progress + '%';

        }


        function animateNumber(id, target) {

            let start = 0;

            const duration = 1500;

            const obj = document.getElementById(id);

            const startTime = performance.now();


            function update(currentTime) {

                const elapsed = currentTime - startTime;

                const progress = Math.min(elapsed / duration, 1);

                const currentNum = Math.floor(progress * target);

                obj.innerText = currentNum.toLocaleString();

                if (progress < 1) requestAnimationFrame(update);

            }

            requestAnimationFrame(update);

        }
